package com.cms.hospice.api;

import com.cms.hospice.model.PricingResult;
import com.fasterxml.jackson.annotation.JsonInclude;

import java.math.BigDecimal;
import java.util.Arrays;
import java.util.List;

/**
 * Outbound JSON response for a hospice claim pricing operation.
 * Maps directly from {@link PricingResult}.
 */
@JsonInclude(JsonInclude.Include.NON_NULL)
public record PricingResponse(
        /** Payment for line item 1 (rev 0651 RHC). */
        BigDecimal payAmt1,

        /** Payment for line item 2 (rev 0652 CHC). */
        BigDecimal payAmt2,

        /** Payment for line item 3 (rev 0655 IRC). */
        BigDecimal payAmt3,

        /** Payment for line item 4 (rev 0656 GIC). */
        BigDecimal payAmt4,

        /** Sum of all line item payments. */
        BigDecimal payAmountTotal,

        /**
         * Return / reason code.
         * "00" = success (pre-FY2016.1).
         * "73"/"74"/"75"/"77" = FY2016.1+ success variants.
         * "10"/"20"/"30"/"40"/"50"/"51" = error.
         */
        String returnCode,

        /** Number of RHC days priced at HIGH rate (FY2016.1+ only). */
        int highRhcDays,

        /** Number of RHC days priced at LOW rate (FY2016.1+ only). */
        int lowRhcDays,

        /** EOL SIA add-on per-day payments for up to 7 days (FY2016.1+ only). */
        List<BigDecimal> eolAddOnDayPayments,

        BigDecimal naAddOnPay1,
        BigDecimal naAddOnPay2) {

    public static PricingResponse from(PricingResult result) {
        List<BigDecimal> eolList = result.eolAddOnDayPayments() != null
                ? Arrays.asList(result.eolAddOnDayPayments())
                : List.of();
        return new PricingResponse(
                result.payAmt1(),
                result.payAmt2(),
                result.payAmt3(),
                result.payAmt4(),
                result.payAmountTotal(),
                result.returnCode(),
                result.highRhcDays(),
                result.lowRhcDays(),
                eolList,
                result.naAddOnPay1(),
                result.naAddOnPay2());
    }
}
